//create readfile.js 
//READ FILE 
var h p =require('h p'); 
var fs = require('fs'); 
h p.createServer(func on (req, res) { 
  fs.readFile('file1.txt', func on(err, data) { 
    res.writeHead(200, {'Content-Type': 'text/html'}); 
    res.write(data); 
    return res.end(); 
  }); 
}).listen(3000);
